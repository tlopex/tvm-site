307  int64_t hash_value = init_hash;

309  hash_value = hash(ty, hash_value, false);

877 /* \brief Allow relax.Var as key in STL tables
878  *
879  * For most Relax expressions, it would be ambiguous whether the
880  * expression should follow reference equality or structural equality.
881  * This is not the case for variables, which do not contain nested
882  * internal structure, and are frequently used as keys in lookup
883  * tables.
884  *
885  * Providing `std::hash` and `std::equal_to` specializations for
886  * `relax::Var` allows it to be used as a key in STL tables. For
887  * `relax::Expr`, the user must specify the type of equality used
888  * (e.g. `std::unordered_set<T, StructuralHash, StructuralEqual>` or
889  * `std::unordered_set<T, ffi::ObjectPtrHash, ffi::ObjectPtrEqual>`).
890  */
891 template <>
892 struct std::hash<tvm::relax::Var> {
893  std::size_t operator()(const tvm::relax::Var& var) const {
894  return tvm::ffi::ObjectPtrHash()(var);
895  }
896 };

902  }
903 };
