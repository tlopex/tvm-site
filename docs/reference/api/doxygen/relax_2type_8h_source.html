401 template <typename T>
402 inline ffi::Optional<T> MatchType(const Expr& expr) {
403  if (!expr.defined()) {
404  return std::nullopt;
405  }
406  using TNode = typename T::ContainerType;
407  if (const TNode* ptr = expr->ty.as<TNode>()) {
408  return ffi::GetRef<T>(ptr);
409  } else {
410  return std::nullopt;
411  }
412 }
