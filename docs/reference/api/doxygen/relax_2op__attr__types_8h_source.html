35 enum OpPatternKind {
36  // Elementwise operation
37  kElemWise = 0,
38  // Broadcasting operator, can always map output axis to the input in order.
39  // for example :code:`out[i, ax1, j, ax2] = input[i, j]`.
40  // Note that the axis need to be in order so transpose is not a bcast operator.
41  kBroadcast = 1,
42  // Injective operator, can always injectively map output axis to a single input axis.
43  // All injective operator can still be safely fused to injective and reduction.
44  kInjective = 2,
45  // Communicative reduction operator.
46  kCommReduce = 3,
47  // Complex operation, can still fuse elemwise operations into its output.
48  // but cannot chain another complex op
49  kOutEWiseFusable = 4,
50  // The pattern for tuple nodes. Can fuse into subsequent injective ops,
51  // but treated specially
52  kTuple = 7,
53  // Opaque operation, cannot fuse anything.
54  kOpaque = 8
55 };
