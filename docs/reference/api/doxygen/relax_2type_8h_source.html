365 template <typename T>
366 inline ffi::Optional<T> MatchType(const Expr& expr) {
367  if (!expr.defined()) {
368  return std::nullopt;
369  }
370  using TNode = typename T::ContainerType;
371  if (const TNode* ptr = expr->ty.as<TNode>()) {
372  return ffi::GetRef<T>(ptr);
373  } else {
374  return std::nullopt;
375  }
376 }
