316  enum Extension {
317  // No extensions enabled
318  kNone = 0,
319 
320  /* When simplifying an inequality, attempt to use scope-based knowns.
321  *
322  * Example:
323  * if_then_else(i<j && j<k, i<k, false) => if_then_else(i<j && j<k, true, false)
324  */
325  kTransitivelyProveInequalities = (1 << 0),
326 
327  /* When simplifying a boolean expression, convert to an AND of ORs
328  * (conjunctive normal form).
329  *
330  * Example:
331  * (a && b) || c => (a || c) && (b || c)
332  */
333  kConvertBooleanToAndOfOrs = (1 << 1),
334 
335  /* When simplifying a boolean AND or a boolean OR, simplify each
336  * branch under the assumption that the other branch does not
337  * already dominate the result. That is, simplify each branch of
338  * (A && B) under the assumption that the other branch is true,
339  * and simplify each branch of (A || B) under the assumption that
340  * the other branch is false.
341  *
342  * Example:
343  * (n < 10) && (n < 5) => (n < 10)
344  * (n < 10) || (n < 5) => (n < 5)
345  */
346  kApplyConstraintsToBooleanBranches = (1 << 2),
347 
348  /* Special handling for expressions `(A+B)*C < (A*B)*D`
349  *
350  * Expressions of the form `(A+B)*C < (A*B)*D` can occur occur
351  * when comparing the number of operations required for two
352  * different orderings in which matrix multiplications can be
353  * performed. Proving or disproving this conditional allows an
354  * optimal order of execution to be selected, even for dynamic
355  * argument shapes.
356  *
357  * The default behavior of `ConstIntBounds` assumes that each term
358  * in an expression is independent, and is insufficient to prove
359  * these inequalities. For example, the maximum value of `(A+B)*C
360  * - (A*B)*D` is determined by taking the maximum value of
361  * `(A+B)*C` and subtracting the minimum value of `(A*B)*D`.
362  * While this algorithm can be applied in all cases, the bound it
363  * provides is looser than strictly required.
364  *
365  * This extension adds a check for this case. When `A`, `B`, `C`,
366  * and `D` are all positive values, as is the case for tensor
367  * shapes, the inequality can be written as `1/A + 1/B < D/C`. If
368  * this inequality holds for the minimum values of `A`, `B`, and
369  * `D`, along with the maximum value of `C`, then the inequality
370  * holds for all values.
371  *
372  * This extension requires little to no performance overhead, and
373  * may be enabled by default in future releases.
374  */
375  kComparisonOfProductAndSum = (1 << 3),
376  };
